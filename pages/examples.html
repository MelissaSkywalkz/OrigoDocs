    Quick recipes

    WMS GetMap

    curl -G 'https://srv-origo01.kommun.skovde.se/geoserver/ows/' \
  --data-urlencode 'service=WMS' \
  --data-urlencode 'request=GetMap' \
  --data-urlencode 'layers=publik:fb_kommunal_fastighet_y' \
  --data-urlencode 'srs=EPSG:3008' \
  --data-urlencode 'bbox=172500,6477600,173500,6478600' \
  --data-urlencode 'width=512' \
  --data-urlencode 'height=512'
